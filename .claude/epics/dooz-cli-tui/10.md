### Cleanup on Exit
```rust
use crossterm::{terminal::LeaveAlternateScreen, execute};

fn cleanup() {
    // Restore terminal state
    crossterm::execute!(std::io::stdout(), LeaveAlternateScreen).ok();
    crossterm::terminal::disable_raw_mode().ok();
}
```

### Error Handling
```rust
fn load_sessions() -> Result<Vec<Session>> {
    let store = SessionStore::new()
        .context("Failed to create session store")?;
    store.list_sessions()
        .context("Failed to load sessions")
}

// Never unwrap in user-facing code
if let Err(e) = save_session(&session) {
    tracing::error!("Failed to save session: {}", e);
    // Continue running, session in memory
}
```

### Panic Handler
```rust
std::panic::set_hook(Box::new(|panic_info| {
    let _ = cleanup();
    eprintln!("Panic: {}", panic_info);
}));
```

## Technical Details

### Message Alignment
```rust
fn view_message(msg: &Message, area: Rect, buf: &mut Buffer) {
    match msg.role {
        MessageRole::User => {
            // Right-aligned, cyan color
            let style = Style::new().cyan();
            render_right_aligned(msg, area, buf, style);
        }
        MessageRole::Assistant => {
            // Left-aligned, white/green color  
            let style = Style::new().green();
            render_left_aligned(msg, area, buf, style);
        }
    }
}
```

### Scroll State
```rust
pub struct ChatModel {
    pub scroll_state: ScrollbarState,
    pub messages: Vec<Message>,
    // ...
}

impl ChatModel {
    fn scroll_up(&mut self) {
        self.scroll_state = self.scroll_state.prev();
    }
    
    fn scroll_down(&mut self) {
        self.scroll_state = self.scroll_state.next();
    }
}
```

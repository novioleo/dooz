### Input Handling
```rust
match action {
    Action::Key(KeyCode::Char(c)) => {
        model.input_buffer.push(c);
    }
    Action::Key(KeyCode::Backspace) => {
        model.input_buffer.pop();
    }
    Action::Key(KeyCode::Enter) => {
        if !model.input_buffer.is_empty() {
            send_message(&model.input_buffer);
            model.input_buffer.clear();
        }
    }
    _ => {}
}
```

### Input Widget
```rust

    .cursor_position(model.cursor_position)

```

### Send Message Flow
```rust
fn send_message(content: &str) -> Option<Action> {
    let message = Message {
        id: Uuid::new_v4(),
        role: MessageRole::User,
        content: content.to_string(),
        timestamp: Utc::now(),
    };
    Some(Action::SendMessage(message))
}
```
